💡 Google Sheets-এ ARRAYFORMULA ব্যবহার করলে পুরো কলামে একটাই ফর্মুলা লিখলেই হয় — প্রতিটি রোতে আলাদা করতে হয় না। Fixed range (A1:A100) এর বদলে A2:A লিখুন — এটা infinite range হিসেবে কাজ করে।

Apps Script — API Code (Extensions → Apps Script → Code.gs) 
function doGet(e) {
  var sheet = SpreadsheetApp.getActive();
  var action = e.parameter.action;

  if (action === 'dashboard') {
    var data = sheet.getSheetByName('Dashboard_Data').getDataRange().getValues();
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', data: data }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  if (action === 'sales') {
    var s = sheet.getSheetByName('Sales_Tracker');
    var rows = s.getDataRange().getValues();
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', data: rows }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  var payload = JSON.parse(e.postData.contents);
  var sheet = SpreadsheetApp.getActive();

  if (payload.action === 'addOrder') {
    var s = sheet.getSheetByName('Sales_Tracker');
    s.appendRow(payload.row);
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'success' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// Auto Timestamp on Edit
function onEdit(e) {
  var sheet = e.source.getActiveSheet();
  var col = e.range.getColumn();
  var row = e.range.getRow();

  if (sheet.getName() === 'Sales_Tracker' && col === 4 && row > 1) {
    sheet.getRange(row, 2).setValue(Utilities.formatDate(new Date(), 'Asia/Dhaka', 'dd-MMM-yyyy'));
    sheet.getRange(row, 3).setValue(Utilities.formatDate(new Date(), 'Asia/Dhaka', 'HH:mm:ss'));
  }
}
Apps Script — Bangladesh Dependent Dropdown (Division → District → Thana) 
function onEdit(e) {
  var sheet = e.source.getActiveSheet();
  var col = e.range.getColumn();
  var row = e.range.getRow();
  var settings = e.source.getSheetByName('Settings');

  // When Division selected (col F = column 6), update District dropdown
  if (sheet.getName() === 'Sales_Tracker' && col === 6 && row > 1) {
    var division = e.range.getValue();
    var districtCol = getDistrictsByDivision(settings, division);
    var rule = SpreadsheetApp.newDataValidation()
      .requireValueInList(districtCol).build();
    sheet.getRange(row, 7).setDataValidation(rule); // col G = District
    sheet.getRange(row, 7).clearContent();
    sheet.getRange(row, 8).clearContent(); // clear Thana
  }

  // When District selected (col G = 7), update Thana dropdown
  if (sheet.getName() === 'Sales_Tracker' && col === 7 && row > 1) {
    var district = e.range.getValue();
    var thanaList = getThanasByDistrict(settings, district);
    var rule = SpreadsheetApp.newDataValidation()
      .requireValueInList(thanaList).build();
    sheet.getRange(row, 8).setDataValidation(rule); // col H = Thana
    sheet.getRange(row, 8).clearContent();
  }
}

function getDistrictsByDivision(settings, division) {
  // Settings Sheet-এ Division-District mapping table থাকবে
  var data = settings.getRange('A:B').getValues();
  var districts = [];
  data.forEach(function(row) {
    if (row[0] === division && row[1]) districts.push(row[1]);
  });
  return districts;
}
